【指定した年月日までのその年での日数】

実習課題9_11 で作成した ex9_11.c に，指定した年月日までの年における日数（その日を含む）を求める
関数 countDays() を実装し，以下のプログラム ex9_12.c を完成させよ．
なお，ex8_12.c は以下の部分に各関数の定義を加えるのみとし，以下に示している部分に
関しては，一切の変更・追記は認めない．
以下の部分が変更・追記されたプログラムは正しく動作しても 0 点とする．

#include<stdio.h>
#include<ctype.h>
#include<string.h>
#include<stdlib.h>

#define BUFSIZE 128

int days[] = { 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
char day_of_week[7][10]={
	"sunday","monday","tuesday", "wednesday", "thursday","friday", "saturday"};

void getInput(char msg[], char n[]);
int isInt(char n[]);
int getYear(void);
int getMonth(void);
int getDay(int y, int m);
int getDays(int y, int m);
int isLeapYear(int y);
int getDayOfWeekForFirstDay(int y);
int countDays(int y, int m, int d);

int main(){
	int year, month, day, dow_first, days_in_year;
	
	year = getYear();
	printf("西暦：%d\n", year);
	month = getMonth();
	printf("月：%d\n", month);
	day = getDay(year, month);
	printf("日：%d\n", day);
	dow_first = getDayOfWeekForFirstDay(year);
	printf("%d年の1月1日の曜日は %s\n", year, day_of_week[dow_first]);
	days_in_year = countDays(year, month, day);
	printf("%d年%d月%d日はその年の%d日目\n", year, month, day, days_in_year);

	return 0;
}
