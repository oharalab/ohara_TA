【ヒストグラムの出力（ファイルからの入力）】

実習課題 6_1で作成した ex6_1.c を，入力データを指定したファイルから読み込むように
変更したプログラム ex6_2.c を作成せよ．
なお，データの読み込みは関数 input_data() を利用するものとし，データの個数は100個とする．
CoursePower 上にある data_ex6_2.txt を用いて動作確認せよ．
また，以下の仕様に従うこと．

#include ⟨stdio.h⟩
#include ⟨stdlib.h⟩

#define bin_size 10
#define max_size 20
#define d_size 100

int input_data(char fname[], int d[]);
void make_hist(int d[], int h[][max_size]);
int print_vhist(int h[][max_size]);

int main(){
	int data[d_size] = {};
	int hist[bin_size][max_size] = {};
	char fname[100];
	
	printf("ファイル名を入力してください：");
	scanf("%s", fname);
	if(input_data(fname, data) == 1)
		return 1;
	make_hist(data, hist);
	if(print_vhist(hist) == 1)
		return 1;

	return 0;
}
