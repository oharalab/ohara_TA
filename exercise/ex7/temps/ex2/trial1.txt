#include<string.h>

    char name[30];
    int age;
    int grade;
    double gpa;
} personal_data;

void clear1(personal_data d);
void clear2(personal_data d[]);

    personal_data data[5] = {"Aoyama Taro", 20, 2, 3.0};
	
    printf("in main¥n");
    print_data(data[0]);

    clear1(data[0]);
	
    printf("in main¥n");
    print_data(data[0]);

    clear2(data);
	
    printf("in main¥n");
    print_data(data[0]);

    return(0);
}

void clear1(personal_data d){
    d.name[0] = '¥0';
    d.age = 0;
    d.grade = 0;
    d.gpa = 0.0;
	
    printf("in clear1¥n");
    print_data(d);
}

    d[0].name[0] = '¥0';
    d[0].age = 0;
    d[0].grade = 0;
    d[0].gpa = 0.0;

    printf("in clear2¥n");
    print_data(d[0]);
}
